176  bool iamSyncer;
177  if ( pthread_mutex_trylock ( &managedMemory::parentalMutex ) == 0 ) {
178  //Could lock
179  iamSyncer = true;
180  } else {
181  //Could not lock. Perhaps, I already have a lock:
182  if ( pthread_equal ( pthread_self(), managedMemory::creatingThread ) ) {
183  iamSyncer = false;//I was called by my parents!
184  } else {
185  //I need to gain the lock:
186  rambrain_pthread_mutex_lock ( &managedMemory::parentalMutex );
187  iamSyncer = true;
188  }
189  }
190  //iamSyncer tells me whether I am the parent thread (not object!)
191 
192  //Set our thread id as the one without locking:
193  if ( iamSyncer ) {

196  }

203  memoryID savedParent = managedMemory::parent;
204  managedMemory::parent = chunk->id;

343  void waitForSwapin() const {
344  //While in this case, we are not the guy who actually enforce the swapin, we never the less have to wait
345  //for the chunk to become ready. This is done in the following way:
346  if ( ! ( chunk->status & MEM_ALLOCATED ) ) { // We may savely check against this as use will be set by other adhereTo thread and cannot be undone as long as calling adhereTo exists
347  rambrain_pthread_mutex_lock ( &managedMemory::defaultManager->stateChangeMutex );
348  //We will burn a little bit of power here, eventually, but this is a very rare case.
349  while ( ! managedMemory::defaultManager->waitForSwapin ( *chunk, true ) ) {};
350  rambrain_pthread_mutex_unlock ( &managedMemory::defaultManager->stateChangeMutex );
351  }
