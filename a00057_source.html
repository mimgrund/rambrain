113  if ( pthread_mutex_trylock ( &managedMemory::parentalMutex ) == 0 ) {
114  //Could lock

116  } else {
117  //Could not lock. Perhaps, I already have a lock:
118  if ( pthread_equal ( pthread_self(), managedMemory::creatingThread ) ) {
119  iamSyncer = false;//I was called by my parents!
120  } else {
121  //I need to gain the lock:
122  rambrain_pthread_mutex_lock ( &managedMemory::parentalMutex );
123  iamSyncer = true;
124  }
125  }
126  //iamSyncer tells me whether I am the parent thread (not object!)

128  //Set our thread id as the one without locking:

279  void waitForSwapin() const {
280  //While in this case, we are not the guy who actually enforce the swapin, we never the less have to wait
281  //for the chunk to become ready. This is done in the following way:
282  if ( ! ( chunk->status & MEM_ALLOCATED ) ) { // We may savely check against this as use will be set by other adhereTo thread and cannot be undone as long as calling adhereTo exists
283  rambrain_pthread_mutex_lock ( &managedMemory::defaultManager->stateChangeMutex );
284  //We will burn a little bit of power here, eventually, but this is a very rare case.
285  while ( ! managedMemory::defaultManager->waitForSwapin ( *chunk, true ) ) {};
286  rambrain_pthread_mutex_unlock ( &managedMemory::defaultManager->stateChangeMutex );
