475 bool cyclicManagedMemory::checkCycle() const
476 {
477  rambrain_pthread_mutex_lock ( &stateChangeMutex );
478  rambrain_pthread_mutex_lock ( &cyclicTopoLock );
479 #ifdef PARENTAL_CONTROL
480  unsigned int no_reg = memChunks.size() - 1;
481 #else
482  unsigned int no_reg = memChunks.size();
483 #endif
484  unsigned int encountered = 0;
485  cyclicAtime *cur = active;
486  cyclicAtime *oldcur;
487 
488  global_bytesize usedBytes = 0;
489  global_bytesize tobef = 0;
490 
491  if ( !cur ) {
492  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
493  rambrain_pthread_mutex_unlock ( &stateChangeMutex );
494  if ( no_reg == 0 && counterActive == NULL ) {
495  return true;
496  } else {
497  errmsg ( "No active element found" );
498  return false;
499  }
500  }
501 
502  bool inActiveOnlySection = ( active->chunk->status == MEM_SWAPPED || active->chunk->status == MEM_SWAPOUT ? false : true );
503  bool inSwapsection = true;
504  bool memerror = false;
505  do {
506  ++encountered;
507  oldcur = cur;
508  cur = cur->next;
509  if ( cur->chunk->status & MEM_ALLOCATED || cur->chunk->status == MEM_SWAPIN ) {
510  usedBytes += cur->chunk->size;
511  } else if ( cur->chunk->status == MEM_SWAPOUT ) {
512  tobef += cur->chunk->size;
513  usedBytes += cur->chunk->size;
514  }
515 
516  if ( oldcur != cur->prev ) {
517  errmsgf ( "Mutual connecion failure at chunks %lu and %lu", oldcur->chunk->id, cur->chunk->id );
518  memerror = true;
519  }
520 
521  if ( inActiveOnlySection ) {
522  if ( oldcur->chunk->status == MEM_SWAPPED || oldcur->chunk->status == MEM_SWAPOUT ) {
523  errmsg ( "Swapped elements in active section!" );
524  memerror = true;
525  }
526  } else {
527  if ( oldcur->chunk->status == MEM_SWAPPED && !inSwapsection ) {
528  errmsg ( "Isolated swapped element block not tracked by counterActive found!" );
529  memerror = true;
530  }
531  if ( oldcur->chunk->status != MEM_SWAPPED && oldcur->chunk->status != MEM_SWAPOUT ) {
532  inSwapsection = false;
533  }
534  }
535 
536  if ( oldcur == counterActive ) {
537  inActiveOnlySection = false;
538  }
539 
540  } while ( cur != active );
541 
542 
543  if ( usedBytes != memory_used ) {
544  errmsgf ( "Used bytes are not counted correctly, claimed %lu but found %lu", memory_used, usedBytes );
545  memerror = true;
546  }
547  if ( tobef != memory_tobefreed ) {
548  errmsgf ( "To-Be-Freed bytes are not counted correctly, claimed %lu but found %lu", memory_used, usedBytes );
549  memerror = true;
550  }
551 
552  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
553  rambrain_pthread_mutex_unlock ( &stateChangeMutex );
554  if ( encountered != no_reg ) {
555  errmsgf ( "Not all elements accessed. %d expected, %d encountered", no_reg, encountered );
556  memerror = true;
557  }
558 
559  if ( memerror ) {
560  return false;
561  } else {
562  return true;
563  }
